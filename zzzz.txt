Test Notification in Tinker:
use App\Models\Article;
use App\Events\ArticleCreated;
use Illuminate\Support\Facades\Event;

// Loop and create articles, then trigger event
collect(range(1,5))->each(function($i) {
    $article = Article::create([
        'title' => "Test Article #{$i}",
        'description' => "This is a test notification {$i}",
        'category_id' => 1,      // Make sure this category exists
        'source' => 'Tinker',
        'url' => "https://example.com/test-article-{$i}",
        'published_at' => now(),
    ]);

    // 2️⃣ Trigger the event
    event(new ArticleCreated($article));

    echo "Event triggered for article: {$article->title}\n";
});
